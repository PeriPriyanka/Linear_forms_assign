\section{Problem}
(Linear forms 2.3) Draw the graphs of the following equations\\
\begin{align}
\myvec{1& 1}\vec{X}=4\\
\myvec{1&-1}\vec{X}=2\\
\myvec{3&-1}\vec{X}=0
\end{align}
\section{Solution}
Consider the equation (1.0.1)
\begin{align}
\myvec{1& 1}\vec{X}=4
\end{align}
where,
\begin{align}
\vec{X}= \myvec{x\\y}
\end{align}
By performing matrix multiplication
\begin{align}
\myvec{1&1}\myvec{x\\y}=4\\
\vec{AB}=\vec{x}+\vec{y}=4
\end{align}
Point Vector
\begin{align}
\vec{AB} =\vec{B}-\vec{A}\\
\vec{AB} =\myvec{4\\0}-\myvec{0\\4}\\
\vec{AB} =\myvec{4\\-4}
\end{align}
Consider the equation (1.0.2)
\begin{align}
\myvec{1& -1}\vec{X}=2
\end{align}
By performing matrix multiplication
\begin{align}
\myvec{1&-1}\myvec{x\\y}=2\\
\vec{CD}=\vec{x}-\vec{y}=2
\end{align}
Point Vector
\begin{align}
\vec{CD} =\vec{D}-\vec{C}\\
\vec{CD} =\myvec{2\\0}-\myvec{0\\-2}\\
\vec{CD} =\myvec{2\\2}
\end{align}
Consider the equation (1.0.3)
\begin{align}
\myvec{3& -1}\vec{X}=0
\end{align}
By performing matrix multiplication
\begin{align}
\myvec{3&-1}\myvec{x\\y}=0\\
\vec{EF}=\vec{3x}-\vec{y}=0
\end{align}
Point Vector
\begin{align}
\vec{EF} =\vec{F}-\vec{E}\\
\vec{EF} =\myvec{0\\0}-\myvec{2\\6}\\
\vec{EF} =\myvec{-2\\-6}
\end{align}
